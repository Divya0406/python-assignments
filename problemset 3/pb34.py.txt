str=(raw_input("enter the strings"))
strlist=str.split(',')
strlist2=[]
for i in strlist:
  if 'e' not in i and 'E' not in i:
    strlist2.append(i)
print "list of words without 'E'",strlist2    
a=len(strlist)
b=len(strlist2)
per=((float(b)/float(a))*100)
print "\n",per,"% of list's word has no 'E'"
